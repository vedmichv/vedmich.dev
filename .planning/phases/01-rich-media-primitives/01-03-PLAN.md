    Frontmatter:
    ```astro
    ---
    /**
     * VvStage — SSR host for VvNode / VvWire / VvPacket primitives.
     *
     * Workflow per render:
     *   1. createStageRegistry() → fresh per-stage Registry with a unique uid
     *   2. pushStage(registry) → module-level singleton for children to find
     *   3. Astro.slots.render('default') → VvNode/VvWire/VvPacket frontmatter runs,
     *      each child calls currentRegistry().registerNode/Wire
     *   4. popStage() (in try/finally)
     *   5. Walk registry.wires → compute d strings via _vv-path.computeWirePath
     *   6. Emit <figure><svg><defs><g><Fragment set:html={slotHtml}/></svg></figure>
     *
     * Coordinate system: SVG user-space 1800×820 by default (D-03). Override via viewBox prop.
     * Aspect-ratio: computed from viewBox so overriding viewBox reshapes the canvas.
     * Per D-04, nodes, wires, and packets all live inside the SAME <svg> — this is the
     * contract that eliminates the CSS offset-path two-coordinate-space bug.
     */
    import {
      createStageRegistry,
      pushStage,
      popStage,
    } from './_vv-registry.ts';
    import { computeWirePath } from './_vv-path.ts';
    import type { Rect } from './_vv-geom.ts';

    interface Props {
      title?: string;
      caption?: string;
      viewBox?: string;
    }

    const { title, caption, viewBox = '0 0 1800 820' } = Astro.props;

    // Parse viewBox "minX minY w h" → derive aspect-ratio for responsive sizing
    const vbParts = viewBox.trim().split(/\s+/).map(Number);
    if (vbParts.length !== 4 || vbParts.some(Number.isNaN)) {
      throw new Error(`<VvStage viewBox="${viewBox}"> — must be "minX minY width height" with 4 numeric values.`);
    }
    const [, , vbW, vbH] = vbParts;
    const aspectRatioStyle = `aspect-ratio: ${vbW} / ${vbH};`;

    // Set up per-stage registry
    const registry = createStageRegistry();
    pushStage(registry);

    let slotHtml: string;
    try {
      slotHtml = await Astro.slots.render('default');
    } finally {
      popStage();
    }

    // Resolve wires → compute d strings
    interface WireDef {
      id: string;
      d: string;
      dashed: boolean;
      tone: string;
      delay: number;
    }
    const wireDefs: WireDef[] = [];
    for (const w of registry.wires.values()) {
      const fromNode = registry.getNode(w.from);
      const toNode = registry.getNode(w.to);
      if (!fromNode || !toNode) {
        const known = [...registry.nodes.keys()].join(', ') || '(none registered)';
        throw new Error(
          `<VvWire from="${w.from}" to="${w.to}"> — unknown node id. ` +
          `Known node ids in this <VvStage>: ${known}.`,
        );
      }
      const fromRect: Rect = {
        cx: fromNode.x + fromNode.w / 2,
        cy: fromNode.y + fromNode.h / 2,
        w: fromNode.w,
        h: fromNode.h,
      };
      const toRect: Rect = {
        cx: toNode.x + toNode.w / 2,
        cy: toNode.y + toNode.h / 2,
        w: toNode.w,
        h: toNode.h,
      };
      wireDefs.push({
        id: w.id,
        d: computeWirePath(fromRect, toRect, { curve: w.curve, via: w.via }),
        dashed: w.dashed ?? false,
        tone: w.tone ?? 'teal',
        delay: w.delay ?? 0,
      });
    }

    const glowId = `glow-${registry.uid}`;
    ---
    ```
