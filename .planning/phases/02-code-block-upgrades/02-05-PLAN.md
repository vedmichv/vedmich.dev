    ---
    // Enhance Shiki-rendered code blocks inside `.prose` with a copy-to-clipboard
    // button. Pairs with `rehype-code-badge.mjs` which wraps every <pre class="shiki">
    // in <figure class="code-block">. This script targets `.code-block > pre` — the
    // figure already provides the toolbar container.
    //
    // UX (D-02, D-07, D-07b): button always visible (icon-only at rest), expands to
    // [icon + label] on .code-block:hover. On successful copy, sets the singleton
    // toast text in BaseLayout (#code-copy-toast) + toggles .is-visible for ~2s.
    // No is-copied intra-button feedback (toast is canonical — RESEARCH.md Pitfall 7).
    //
    // Preserves: Clipboard API + textarea fallback, aria-label, bilingual labels.

    import { Icon } from 'astro-icon/components';

    interface Props {
      locale?: 'en' | 'ru';
    }

    const { locale = 'en' } = Astro.props;

    const labels = {
      en: { copy: 'Copy', toastCopied: 'Copied to clipboard' },
      ru: { copy: 'Копировать', toastCopied: 'Скопировано в буфер' },
    } as const;

    const l = labels[locale];
    ---

    <!-- SSR template: inert at page load, cloned at runtime by the script below.
         Using astro-icon's <Icon> here keeps the Lucide SVG canonical (not hardcoded). -->
    <template id="code-copy-btn-template">
      <button type="button" class="code-copy-btn" aria-label={l.copy}>
        <Icon name="lucide:copy" width="14" height="14" aria-hidden="true" />
        <span class="code-copy-btn-label">{l.copy}</span>
      </button>
    </template>

    <script
      is:inline
      data-copy-label={l.copy}
      data-toast-copied-label={l.toastCopied}
    >
      (() => {
        const script = document.currentScript;
        const COPY_LABEL = script?.dataset.copyLabel ?? 'Copy';
        const TOAST_COPIED = script?.dataset.toastCopiedLabel ?? 'Copied';

        // Clone the SSR template once per page (shared across all buttons).
        const template = document.getElementById('code-copy-btn-template');
        if (!(template instanceof HTMLTemplateElement)) return;

        const pres = document.querySelectorAll('.prose .code-block > pre');
        pres.forEach((pre) => {
          const figure = pre.parentElement;
          if (!figure || !figure.classList.contains('code-block')) return;
          // Skip if a copy button already exists (hot-reload or double-run safety).
          if (figure.querySelector('.code-copy-btn')) return;

          const frag = template.content.cloneNode(true);
          const btn = frag.querySelector('.code-copy-btn');
          if (!(btn instanceof HTMLButtonElement)) return;
          btn.setAttribute('aria-label', COPY_LABEL);
          figure.appendChild(btn);

          btn.addEventListener('click', async () => {
            const code = pre.querySelector('code');
            const text = code ? code.innerText : pre.innerText;

            try {
              await navigator.clipboard.writeText(text);
            } catch {
              // Fallback: off-screen textarea selection + execCommand.
              const ta = document.createElement('textarea');
              ta.value = text;
              ta.style.position = 'fixed';
              ta.style.top = '-1000px';
              document.body.appendChild(ta);
              ta.select();
              try { document.execCommand('copy'); } catch { /* noop */ }
              document.body.removeChild(ta);
            }

            // Toast feedback — singleton in BaseLayout.astro.
            const toast = document.getElementById('code-copy-toast');
            if (toast) {
              toast.textContent = TOAST_COPIED;
              toast.classList.add('is-visible');
              // Clear any pending hide from a previous copy so double-clicks
              // reset the 2s window (last-write-wins per D-07c).
              if (toast.__hideTimer) clearTimeout(toast.__hideTimer);
              toast.__hideTimer = setTimeout(() => {
                toast.classList.remove('is-visible');
              }, 2000);
            }
          });
        });
      })();
    </script>

    <style is:global>
      /* Copy button baseline — always visible, icon-only at rest. */
      .prose .code-block {
        /* Ensure figure is the positioning context for the absolute button. */
        position: relative;
      }

      .prose .code-copy-btn {

    </style>
